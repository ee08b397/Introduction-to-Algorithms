\textbf{Exercise 3.1-6:}\\
$
\because T(n) \le O(g(n))\ and\ T(n) \ge \Omega(g(n))\\
\therefore T(n) \ge \Omega(g(n)) \ge O(g(n))\\
\therefore O(g(n)) \le T(n) \le O(g(n))\\
\therefore T(n) = O(g(n))\\
\therefore Similarly\ T(n) = \Omega(g(n))\\
\therefore T(n) = \Theta(g(n))\\
$
